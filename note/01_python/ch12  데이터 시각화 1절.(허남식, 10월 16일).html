# 2행 2열 서브플롯(for)
func=[np.sin,np.cos,np.tanh, lambda x:np.sin(x)*np.cos(x)]
fmt=['r', 'c--','#00AA00','r-.']
fig,axes=plt.subplots(2,2,figsize=(7,4))
list(enumerate(axes.flat))
for i, ax in enumerate(axes.flat):
    ax.plot(x,func[i](x),fmt[i])
